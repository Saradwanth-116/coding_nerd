#include <stdio.h>
int main()
{
	int b;
	printf("enter size of array");
	scanf("%d",&b);
	int arr[b];
	for(int i=0; i<b; i++) {
		printf("enter element for the array: ");
		scanf("%d",&arr[i]);
	}
	for(int i=0; i<b; i++) {
		printf("%d",arr[i]);
	}
	for(int i=0; i<b; i++) {
		if(arr[i]==arr[b-i-1]) {
			if(i!=(b-i-1))
			{
				printf("enter array of different element");
				return 0;

			}
		}
	}
	for (int i=0; i<b-1; i++) {
		for (int j=0; j<b-i-1; j++) {
			if(arr[j]>arr[j+1]) {
				int temp=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
			}
		}
	}
	printf("\n");
	for(int i=0; i<b; i++) {
		printf("%d",arr[i]);
	}
	int d;
printf("enter an element to search");
scanf("%d",&d);
int low=0;
int high=b-1;
while(low<=high){
 int mid=low+(high-low)/2;    
    if(arr[mid]==d){
        printf("element is found at index position %d\n",mid);
        return 0;
    }
    else if(d<arr[mid]){
        high=mid-1;
    }
    else{
        low=mid+1;
    }

}
return 0;
}
